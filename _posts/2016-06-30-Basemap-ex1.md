```python
map = Basemap(
```

```python
map.drawcoastlines()
```

```python
map.drawmapboundary()

for i in df['IP']:
    try:
        loc = read.get(i)['location']
        x, y = map(loc.get('longitude'), loc.get('latitude'))
        map.plot(x, y, 'bo', markersize=11)
    except Exception as e:
        print("%s: %s" % (i, e))
```
